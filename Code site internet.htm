// Function to calculate the total price and return summary details

function calculateTotalPrice() {

let basePrice = 0;

const selectedItems = [];

 

// Get all checkboxes with a data-price attribute

const checkboxes = document.querySelectorAll('input[type="checkbox"][data-price]');

 

checkboxes.forEach(checkbox => {

if (checkbox.checked && !checkbox.disabled) {

const price = parseFloat(checkbox.dataset.price);

let itemPrice = price;

let itemName = checkbox.value; // Use simplified value from the value attribute

 

// Get the full description text for the summary

// Look for the span that is the nextElementSibling of the checkbox

const spanElement = checkbox.nextElementSibling;

const displayValue = spanElement ? spanElement.textContent : checkbox.value; // Fallback to value if span not found

 

let quantity = 1;

 

if (checkbox.name === 'option_musique_instrumentale') {

const quantityInput = document.querySelector('input[name="qty_parties_instrumentales"]');

quantity = quantityInput ? parseInt(quantityInput.value) : 0;

itemPrice = price * quantity;

itemName = `${displayValue} (x${quantity})`; // Use full text for summary

} else if (checkbox.name === 'option_arrangement_fournie_instrumentale') {

const quantityInput = document.querySelector('input[name="qty_parties_instrumentales_fournie"]');

quantity = quantityInput ? parseInt(quantityInput.value) : 0;

itemPrice = price * quantity;

itemName = `${displayValue} (x${quantity})`; // Use full text for summary

} else {

itemName = displayValue; // Use full text for other checkboxes

}

 

if (quantity > 0 || !['option_musique_instrumentale', 'option_arrangement_fournie_instrumentale'].includes(checkbox.name)) { // Only add if quantity is greater than 0 for instrumental parts or if it's not an instrumental option

basePrice += itemPrice;

selectedItems.push({

name: itemName,

price: itemPrice

});

}

}

});

 

let finalPrice = basePrice;

let discountAmount = 0;

const applyDiscountCheckbox = document.getElementById('applyDiscount');

// Apply discount only if the checkbox is checked AND a justification file is present

if (applyDiscountCheckbox && applyDiscountCheckbox.checked && justificationFiles.length > 0) {

discountAmount = basePrice * 0.15;

finalPrice = basePrice - discountAmount;

}

 

// Update the main total display

const totalPriceElement = document.getElementById('totalPrice');

if (totalPriceElement) {

totalPriceElement.textContent = finalPrice.toFixed(2);

}

 

// Return details for summary rendering

return {

basePrice: basePrice,

discountAmount: discountAmount,

finalPrice: finalPrice,

selectedItems: selectedItems

};

}

// Function to update the quantity displayed and the hidden input value

function updateQuantity(displayId, hiddenInputName, delta) {

const displayElement = document.getElementById(displayId);

const hiddenInput = document.querySelector(`input[name="${hiddenInputName}"]`);

 

if (!displayElement || !hiddenInput) {

console.error(`Missing elements for quantity update: displayId=${displayId}, hiddenInputName=${hiddenInputName}`);

return;

}

 

let currentQuantity = parseInt(hiddenInput.value);

let newQuantity = currentQuantity + delta;

 

// Ensure quantity doesn't go below zero

if (newQuantity < 0) {

newQuantity = 0;

}

 

hiddenInput.value = newQuantity;

displayElement.textContent = newQuantity;

 

// If the quantity becomes 0 and the corresponding checkbox is checked, uncheck it

if (newQuantity === 0) {

let relatedCheckbox;

if (hiddenInputName === 'qty_parties_instrumentales') {

relatedCheckbox = document.querySelector('input[name="option_musique_instrumentale"]');

} else if (hiddenInputName === 'qty_parties_instrumentales_fournie') {

relatedCheckbox = document.querySelector('input[name="option_arrangement_fournie_instrumentale"]');

}

if (relatedCheckbox && relatedCheckbox.checked) {

relatedCheckbox.checked = false;

}

} else if (newQuantity > 0) {

// If quantity becomes > 0, and the corresponding checkbox is not checked, check it

let relatedCheckbox;

if (hiddenInputName === 'qty_parties_instrumentales') {

relatedCheckbox = document.querySelector('input[name="option_musique_instrumentale"]');

} else if (hiddenInputName === 'qty_parties_instrumentales_fournie') {

relatedCheckbox = document.querySelector('input[name="option_arrangement_fournie_instrumentale"]');

}

if (relatedCheckbox && !relatedCheckbox.checked && !relatedCheckbox.disabled) {

relatedCheckbox.checked = true;

}

}

renderSummary(); // Recalculate total and render summary

}

// Generic setup for drop zones (visual only)

function setupDropZone(dropZoneId, fileInputId, fileListId, isJustificationZone = false) {

const dropZone = document.getElementById(dropZoneId);

const fileInput = document.getElementById(fileInputId);

const fileList = document.getElementById(fileListId);

 

if (!dropZone || !fileInput || !fileList) return;

 

dropZone.addEventListener('click', () => fileInput.click());

 

dropZone.addEventListener('dragover', (e) => {

e.preventDefault();

dropZone.classList.add('hover');

});

 

dropZone.addEventListener('dragleave', (e) => {

e.preventDefault(); // Prevent default to allow drop outside

dropZone.classList.remove('hover');

});

 

dropZone.addEventListener('drop', (e) => {

e.preventDefault();

dropZone.classList.remove('hover');

const files = e.dataTransfer.files;

handleFiles(files, fileList, isJustificationZone);

});

 

fileInput.addEventListener('change', (e) => {

const files = e.target.files;

handleFiles(files, fileList, isJustificationZone);

});

}

 

function handleFiles(files, fileListElement, isJustificationZone) {

if (!fileListElement) return;

 

fileListElement.innerHTML = ''; // Clear previous list

const fileNames = [];

if (files.length > 0) {

const ul = document.createElement('ul');

ul.className = 'list-disc list-inside mt-2';

Array.from(files).forEach(file => {

const li = document.createElement('li');

li.textContent = file.name;

ul.appendChild(li);

fileNames.push(file.name);

});

fileListElement.appendChild(ul);

} else {

fileListElement.textContent = 'Aucun fichier slectionn.';

}

 

if (isJustificationZone) {

justificationFiles = fileNames;

}

renderSummary(); // Recalculate total and render summary after file change

}

 

// Function to generate PDF summary

async function generatePdfSummary() {

const loadingIndicator = document.getElementById('loadingIndicator');

if (loadingIndicator) loadingIndicator.style.display = 'block';

 

// Give a small delay to ensure loading indicator is visible

await new Promise(resolve => setTimeout(resolve, 50));

 

const summaryElement = document.getElementById('summaryContent');

const priceDisplayElement = document.querySelector('.price-display'); // Get the total price display

const summarySubtotal = document.getElementById('summarySubtotal').textContent;

const summaryDiscountAmount = document.getElementById('summaryDiscountAmount').textContent;

const summaryFinalTotal = document.getElementById('summaryFinalTotal').textContent;

const summaryDiscountRowHidden = document.getElementById('summaryDiscountRow').classList.contains('hidden');

 

// Create a temporary div to combine summary content and total prices for PDF

const tempDiv = document.createElement('div');

tempDiv.style.padding = '20px';

tempDiv.style.backgroundColor = '#ffffff'; // White background for PDF

tempDiv.style.fontFamily = 'Inter, sans-serif';

tempDiv.style.fontSize = '12px';

 

tempDiv.innerHTML = `

<h1 style="font-size: 24px; font-weight: bold; text-align: center; margin-bottom: 10px;">Devis - Services crations textuelles et musicales</h1>

<p style="text-align: center; margin-bottom: 20px;">Rcapitulatif de votre commande</p>

<div style="border: 1px solid #e2e8f0; border-radius: 8px; padding: 15px; margin-bottom: 20px;">

${summaryElement.innerHTML}

</div>

<div style="text-align: right; font-size: 14px; font-weight: bold; margin-top: 20px;">

<p>Sous-total : ${summarySubtotal}</p>

${summaryDiscountRowHidden ? '' : `<p style="color: #dc2626;">Rduction (15%) : ${summaryDiscountAmount}</p>`}

<p style="font-size: 18px; color: #4338ca; margin-top: 10px;">Total final : ${summaryFinalTotal}</p>

</div>

`;

document.body.appendChild(tempDiv); // Append temporarily to body for html2canvas to render it correctly

 

try {

const canvas = await html2canvas(tempDiv, { scale: 2 }); // Scale for better resolution

const imgData = canvas.toDataURL('image/png');

const { jsPDF } = window.jspdf;

const doc = new jsPDF('p', 'mm', 'a4'); // Portrait, millimeters, A4 size

 

const imgWidth = 210; // A4 width in mm

const pageHeight = 297; // A4 height in mm

const imgHeight = canvas.height * imgWidth / canvas.width;

let heightLeft = imgHeight;

let position = 0;

 

doc.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);

heightLeft -= pageHeight;

 

while (heightLeft >= 0) {

position = heightLeft - imgHeight;

doc.addPage();

doc.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);

heightLeft -= pageHeight;

}

 

doc.save('Devis_Services.pdf');

} catch (error) {

console.error("Erreur lors de la gnration du PDF:", error);

// Use a custom modal instead of alert()

showCustomModal("Une erreur est survenue lors de la gnration du PDF.");

} finally {

if (loadingIndicator) loadingIndicator.style.display = 'none';

document.body.removeChild(tempDiv); // Remove the temporary div

}

}
